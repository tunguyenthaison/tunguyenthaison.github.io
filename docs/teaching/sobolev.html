Let

\begin{equation*} \mathbf{V}(\mathbf{x}) = 2-\sin (2\pi x) - \sin (2\pi y), \qquad \mathbf{x}= (x,y)\in \mathbb {T}^2. \end{equation*}

We are interested in estimating, for \(\mu {\gt}0\) and \(\gamma {\gt} 0\) the Sobolev norm \(H^{2+s}(\mathbb {T}^2)\) of the function

\begin{equation*} \mathbf{K}_\mu (\mathbf{x}) = (\mu + \mathbf{V}(\mathbf{x}))^{{-1/2}}, \qquad \mathbf{x}\in \mathbb {T}^2. \end{equation*}

1 Evaluate the \(3rd\)-derivative

We want to compute

\begin{align*} \partial ^2_{x_ix_j} \mathbf{K}_\mu (\mathbf{x}) & = \frac{3\gamma ^2}{4}\left(\mu +\mathbf{V}(\mathbf{x})^\gamma \right)^{-5/2} \left(\gamma \mathbf{V}^{\gamma -1} \mathbf{V}_{x_i}\right) \left(\gamma \mathbf{V}^{\gamma -1} \mathbf{V}_{x_j}\right) \\ & -\frac{1}{2}(\mu + \mathbf{V}^\gamma )^{-3/2} \left( \gamma (\gamma -1)\mathbf{V}^{\gamma -2} \mathbf{V}_{x_i}\mathbf{V}_{x_j} + \gamma \mathbf{V}^{\gamma -1} \mathbf{V}_{x_ix_j} \right). \end{align*}

We have

\begin{align*} \partial _{x_k}\Big(\partial ^2_{x_ix_j} \mathbf{K}_\mu (\mathbf{x}) \Big) = I_1 + I_2 + I_3 + I_4, \end{align*}

where

\begin{align*} I_1 & = -\frac{15\gamma ^2}{8} (\mu +\mathbf{V}^\gamma )^{-7/2} \prod _{i,j,k} \left(\gamma \mathbf{V}^{\gamma -1}\mathbf{V}_{x_i}\right) \\ I_2 & = \frac{3\gamma ^2}{4}(\mu +\mathbf{V}^\gamma )^{-5/2} \frac{\partial }{\partial _{x_k}} \left(\gamma \mathbf{V}^{\gamma -1}\mathbf{V}_{x_i}\right) \left(\gamma \mathbf{V}^{\gamma -1}\mathbf{V}_{x_j}\right)\\ I_3 & = \frac{3}{4}(\mu +\mathbf{V}^\gamma )^{-5/2} \left( \gamma \mathbf{V}^{\gamma -1}\mathbf{V}_{x_k} \right) \left( \gamma (\gamma -1)\mathbf{V}^{\gamma -2}\mathbf{V}_{x_i}\mathbf{V}_{x_j} + \gamma \mathbf{V}^{\gamma -1} \mathbf{V}_{x_ix_j} \right) \\ I_4 & = -\frac{1}{2}(\mu +\mathbf{V}^\gamma )^{-3/2} \frac{\partial }{\partial x_k} \left( \gamma (\gamma -1)\mathbf{V}^{\gamma -2} \mathbf{V}_{x_i}\mathbf{V}_{x_j} + \gamma \mathbf{V}^{\gamma -1} \partial ^2_{x_ix_j}\mathbf{V}\right). \end{align*}

We only need to differentiate \(I_2\) and \(I_4\).

1.1 Compute \(I_1\)

We have

\begin{align*} \fbox{$\displaystyle I_1 \leq \frac{ |\mathbf{x}-\mathbf{x}_0|^{6\gamma -3} }{ (\mu +|\mathbf{x}-\mathbf{x}_0|^{2\gamma })^{7/2} } $}. \end{align*}

1.2 Compute \(I_2\)

We have

\begin{align*} & \frac{\partial }{\partial _{x_k}} \left(\gamma \mathbf{V}^{\gamma -1}\mathbf{V}_{x_i}\right) \left(\gamma \mathbf{V}^{\gamma -1}\mathbf{V}_{x_j}\right) = \\ & \qquad \Big[ \gamma (\gamma -1)\mathbf{V}^{\gamma -2} \mathbf{V}_{x_i}\mathbf{V}_{x_k} + \gamma \mathbf{V}^{\gamma -1} \mathbf{V}_{x_ix_k} \Big] \left(\gamma \mathbf{V}^{\gamma -1}\mathbf{V}_{x_j}\right) \color {red} \sim \left(|\mathbf{x}-\mathbf{x}_0|^{2\gamma -2} + |\mathbf{x}-\mathbf{x}_0|^{2\gamma -2}\right)|\mathbf{x}-\mathbf{x}_0|^{2\gamma -1} \color {black} \\ & \qquad \qquad + \\ & \qquad \Big[ \gamma (\gamma -1)\mathbf{V}^{\gamma -2} \mathbf{V}_{x_j}\mathbf{V}_{x_k} + \gamma \mathbf{V}^{\gamma -1}\mathbf{V}_{x_jx_k} \Big] \left(\gamma \mathbf{V}^{\gamma -1}\mathbf{V}_{x_i}\right). \end{align*}

Therefore

\begin{align*} I_2 & = \frac{3\gamma ^2}{4}(\mu +\mathbf{V}^\gamma )^{-5/2} \\ & \left( \Big[ \gamma (\gamma -1)\mathbf{V}^{\gamma -2}\mathbf{V}_{x_i} + \gamma \mathbf{V}^{\gamma -1} \mathbf{V}_{x_ix_k} \Big] \left(\gamma \mathbf{V}^{\gamma -1}\mathbf{V}_{x_j}\right) + \Big[ \gamma (\gamma -1)\mathbf{V}^{\gamma -2} \mathbf{V}_{x_j} + \gamma \mathbf{V}^{\gamma -1} \mathbf{V}_{x_jx_k} \Big] \left(\gamma \mathbf{V}^{\gamma -1}\mathbf{V}_{x_i}\right) \right) \end{align*}

And thus

\begin{align*} I_2 \leq \frac{ |\mathbf{x}-\mathbf{x}_0|^{2\gamma -2} \cdot |\mathbf{x}-\mathbf{x}_0|^{2\gamma -1} }{(\mu +|\mathbf{x}-\mathbf{x}_0|^{2\gamma })^{5/2}} = \fbox{$ \displaystyle \frac{ |\mathbf{x}-\mathbf{x}_0|^{4\gamma -3} }{ (\mu +|\mathbf{x}-\mathbf{x}_0|^{2\gamma })^{5/2} } $} \end{align*}

1.3 Compute \(I_3\)

We have

\begin{align*} I_3 \leq \frac{ |\mathbf{x}-\mathbf{x}_0|^{2\gamma -1} \cdot \Big( |\mathbf{x}-\mathbf{x}_0|^{2\gamma -2} + |\mathbf{x}-\mathbf{x}_0|^{2\gamma -2} \Big) }{ (\mu +|\mathbf{x}-\mathbf{x}_0|^{2\gamma })^{5/2} } = \fbox{$ \displaystyle \frac{ |\mathbf{x}-\mathbf{x}_0|^{4\gamma -3} }{ (\mu +|\mathbf{x}-\mathbf{x}_0|^{2\gamma })^{5/2} } $}. \end{align*}

1.4 Compute \(I_4\)

We have

\begin{align*} I_4 & = -\frac{1}{2}(\mu +\mathbf{V}^\gamma )^{-3/2} \left( \gamma (\gamma -1) \sum _{i,j,k} \mathbf{V}^{\gamma -2} \mathbf{V}_{x_ix_j}\mathbf{V}_{x_k} + \gamma \mathbf{V}^{\gamma -1} \mathbf{V}_{x_ix_jx_k} + \gamma (\gamma -1)(\gamma -2)\mathbf{V}^{\gamma -3} \mathbf{V}_{x_i}\mathbf{V}_{x_j}\mathbf{V}_{x_k} \right). \end{align*}

Therefore

\begin{align*} I_4 \leq \frac{ |\mathbf{x}-\mathbf{x}_0|^{2\gamma -3} + |\mathbf{x}-\mathbf{x}_0|^{2\gamma -1} + |\mathbf{x}-\mathbf{x}_0|^{2\gamma -3} }{ (\mu +|\mathbf{x}-\mathbf{x}_0|^{2\gamma })^{3/2} } \leq \fbox{$\displaystyle \frac{|\mathbf{x}-\mathbf{x}_0|^{2\gamma -3}}{(\mu +|\mathbf{x}-\mathbf{x}_0|^{2\gamma })^{3/2}} + \frac{|\mathbf{x}-\mathbf{x}_0|^{2\gamma -1}}{(\mu +|\mathbf{x}-\mathbf{x}_0|^{2\gamma })^{3/2}} $}. \end{align*}

1.5 Putting everything together

We obtain

\begin{align*} I_1+I_2+I_3+I_4 \leq \frac{ |\mathbf{x}-\mathbf{x}_0|^{6\gamma -3} }{ (\mu +|\mathbf{x}-\mathbf{x}_0|^{2\gamma })^{7/2} } + \frac{ |\mathbf{x}-\mathbf{x}_0|^{4\gamma -3} }{ (\mu +|\mathbf{x}-\mathbf{x}_0|^{2\gamma })^{5/2} } + \frac{ |\mathbf{x}-\mathbf{x}_0|^{2\gamma -3} }{ (\mu +|\mathbf{x}-\mathbf{x}_0|^{2\gamma })^{3/2} } + \frac{ |\mathbf{x}-\mathbf{x}_0|^{2\gamma -1} }{ (\mu +|\mathbf{x}-\mathbf{x}_0|^{2\gamma })^{3/2}. } \end{align*}

Therefore, we have

\begin{align*} & \int _{\mathbb {T}^2} |D^3\mathbf{K}_\mu (\mathbf{x})|^2\; d\mathbf{x}\\ & \qquad \leq C \left( 1 + \int _0^1 \frac{r^{12\gamma -5}}{(\mu +r^{2\gamma })^7}\; dr + \int _0^1 \frac{r^{8\gamma -5}}{(\mu +r^{2\gamma })^5}\; dr + \int _0^1 \frac{r^{4\gamma -5}}{(\mu +r^{2\gamma })^3}\; dr + \int _0^1 \frac{r^{4\gamma -1}}{(\mu +r^{2\gamma })^3}\; dr \right)\\ & \qquad \leq C \left( 1 + \mu ^{-1-\frac{2}{\gamma }}\int _0^{\mu ^{-\frac{1}{2\gamma }}} \frac{s^{12\gamma -5}\; ds}{(1+s^{2\gamma })^7} + \mu ^{-1-\frac{2}{\gamma }}\int _0^{\mu ^{-\frac{1}{2\gamma }}} \frac{s^{8\gamma -5}\; ds}{(1+s^{2\gamma })^5} + \mu ^{-1-\frac{2}{\gamma }}\int _0^{\mu ^{-\frac{1}{2\gamma }}} \frac{s^{4\gamma -5}\; ds}{(1+s^{2\gamma })^3} \right. \\ & \qquad \qquad \qquad \qquad \qquad \qquad \qquad \qquad \qquad \qquad \qquad \qquad \qquad \left. + \mu ^{-1}\int _0^{\mu ^{-\frac{1}{2\gamma }}} \frac{s^{4\gamma -1}\; ds}{(1+s^{2\gamma })^3} \right) \\ & \qquad \leq C\left(1 + \mu ^{-1-\frac{2}{\gamma }} + \mu ^{-1}\right). \end{align*}
